## 四、技术实现细节

### 4.1 RAG 配置文件更新

```python
# .rag/config.py 新增

class ParagraphIndex:
    """段落级索引"""
    
    def __init__(self):
        self.documents = {}  # 文档元数据
        self.paragraphs = {}  # 段落数据
        self.inverted_index = defaultdict(list)  # 倒排索引
        self.mention_stats = defaultdict(lambda: defaultdict(int))  # 提及统计
    
    def index_with_paragraphs(self, filepath: str, content: str, year: int):
        """按段落索引文档"""
        doc_id = self._get_doc_id(filepath)
        
        # 段落分割（保留空白行分隔）
        paragraphs = [p.strip() for p in content.split('\n\n') if p.strip()]
        
        for idx, text in enumerate(paragraphs):
            para_id = f"{doc_id}/p{idx}"
            keywords = self._extract_keywords(text)
            
            self.paragraphs[para_id] = {
                "doc_id": doc_id,
                "para_idx": idx,
                "text": text[:500],  # 限制长度
                "keywords": keywords,
                "year": year
            }
            
            # 更新倒排索引
            for kw in keywords:
                self.inverted_index[kw].append(para_id)
                self.mention_stats[kw][doc_id] += 1
    
    def search_timeline(self, keyword: str) -> list:
        """搜索某关键词的时间线"""
        results = []
        for para_id in self.inverted_index.get(keyword, []):
            para = self.paragraphs[para_id]
            doc = self.documents[para["doc_id"]]
            results.append({
                "year": para["year"],
                "doc": doc["source"],
                "para_idx": para["para_idx"],
                "text": para["text"],
                "jump_url": f"{doc['source']}#p{para['para_idx']}"
            })
        
        # 按年份分组
        return self._group_by_year(results)
```

### 4.2 生成时间线的 Obsidian 链接

```python
def generate_obsidian_link(doc_path: str, para_idx: int, anchor_text: str = "") -> str:
    """生成 Obsidian 内部链接"""
    # 例如：[[1987-letter|内在价值定义]]
    filename = Path(doc_path).stem  # "1987-letter-zh"
    return f"[[{filename}#{para_idx}|{anchor_text}]]"

def generate_markdown_link(doc_path: str, para_idx: int) -> str:
    """生成 Markdown 链接"""
    return f"[查看原文 →](./{doc_path}#p{para_idx})"
```
